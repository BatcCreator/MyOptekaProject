using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using System.Data.SqlClient;
namespace Demo
{
    public partial class Form1 : Form
    {
        public string connect = "Data Source=(localdb)\\MSSQLLocalDB;database=demo;Integrated Security=True;TrustServerCertificate=True";
        public Form1()
        {
           
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {
             string login = textBox1.Text;
             string password = textBox2.Text;
            SqlConnection con = new SqlConnection(connect);
            con.Open();
            string zapros = "SELECT * FROM users WHERE login=@login AND password=@password";
            SqlCommand cmd = new SqlCommand(zapros, con);
            cmd.Parameters.AddWithValue("@login", login);
            cmd.Parameters.AddWithValue("@password", password);
            SqlDataReader read = cmd.ExecuteReader();
            if (login == "" || password == "")
            {
                MessageBox.Show("пустые поля", "пустые поля", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }
            else
            {
                if (read.Read())
                {
                    MessageBox.Show("добро пожаловать", "молодец", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    Form2 f = new Form2();
                    f.Show();
                    this.Hide();
                }


                else
                {
                    MessageBox.Show("Пароль неверный", "не молодец", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                }
            }
            
            
        }
    }
}
